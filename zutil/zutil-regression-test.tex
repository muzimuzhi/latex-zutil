%%
%% counting errors and passed checks
%%

% The initial aim here is to count number of \errmessage errors thrown
% in a \TEST, between tests, and even in between \OMIT ... \TIMO pairs,
% thus provide a more readable view for log-based error-rich tests.
%
% Expandable errors (by using \msg_expandable_error:nn and friends) are
% only accounted if thrown in top-levels of a test file or a test.
%
% Engine errors like "! Undefined control sequence" are NOT patchable,
% thus not credited.
%
% Alternatives:
%
% The l3build config variable "recordstatus = true" records exit code(s)
% of _all_ check runs, thus is more time consuming.
% https://github.com/latex3/l3build/issues/398
%
% Checks issued by the expandable \PASSED are counted too.
%
% In expandable contexts, the counts are stored in a l3flag variable.
%
% end-of-test logging [n > 0]
%     [test] <n> error(s)
%     [test] <n> check(s)
% end-of-file logging [n >= 0]
%     ***************
%     [test] <n> error(s)
%     [test] <n> check(s)

\socket_new_plug:nnn { zutil/test/count-errors } { normal }
  {
    \int_gincr:N \g_zutil_test_errors_cnt_int
  }
\socket_assign_plug:nn { zutil/test/count-errors } { normal }
% No group level checking is performed on expandable errors, as errors are
% always logged and it's not controlled by tests in which group does an
% expandable error occur.
\socket_new_plug:nnn { zutil/test/count-exp-errors } { normal }
  {
    \flag_raise:N \l_zutil_test_exp_errors_cnt_flag
  }
\socket_assign_plug:nn { zutil/test/count-exp-errors } { normal }

\cs_new_protected:Npn \zutil_test_count_errors:
  {
    \socket_use:n { zutil/test/count-errors }
  }
\cs_new:Npn \zutil_test_count_exp_errors:
  {
    \socket_use_expandable:n { zutil/test/count-exp-errors }
  }

% end-of-file logging
\hook_gput_code:nnn { zutil/test/END } { . }
  {
    % \OMIT ... \TIMO are not nestable, so unpaired \TIMO is valid use.
    \TIMO
    % emulate output of l3build "recordstatus = true"
    \iow_term:n { *************** }
    % log error counts, always
    \zutil_test_add_expandable_errors_cnt:
    \zutil_test_log_error_counts:n { \g_zutil_test_errors_cnt_int }
    % log check counts, always
    \zutil_test_log_check_counts:n { \g_zutil_test_check_cnt_int }
  }

% before-test logging
\hook_gput_code:nnn { zutil/test/TEST/before } { . }
  {
    % reset error counts
    \zutil_test_add_expandable_errors_cnt:
    \int_gset_eq:NN \g_zutil_test_errors_tmp_int \g_zutil_test_errors_cnt_int
    \flag_if_raised:NT \l_zutil_test_exp_errors_cnt_flag
      {
        \zutil_test_type:e
          {
            \flag_height:N \l_zutil_test_exp_errors_cnt_flag \c_space_tl
            expandable~errors~raised~between~tests.
          }
        \flag_clear:N \l_zutil_test_exp_errors_cnt_flag
      }
    % reset check counts
    \flag_if_raised:NT \l_zutil_test_checks_cnt_flag
      {
        \zutil_test_type:e
          {
            \flag_height:N \l_zutil_test_checks_cnt_flag \c_space_tl
            checks~used~between~tests.
          }
        \flag_clear:N \l_zutil_test_checks_cnt_flag
      }
  }

\hook_gput_code:nnn { zutil/test/TEST/begin } { . }
  {
    \int_gset_eq:NN \g_zutil_test_group_level_int \currentgrouplevel
    % enable checking commands
    \cs_set_eq:NN \PASSED \__zutil_test_PASSED:
  }
\cs_new:Npn \zutil_test_type:e #1
  {
    \iow_term:e { [test]~ #1 }
  }

% end-of-test logging
\hook_gput_code:nnn { zutil/test/TEST/end } { . }
  {
    % check group and if levels
    \zutil_test_check_group_level:n { \g_zutil_test_group_level_int }
    % log error counts
    \zutil_test_add_expandable_errors_cnt:
    \int_compare:nNnF
      { \g_zutil_test_errors_cnt_int } = { \g_zutil_test_errors_tmp_int }
      {
        \zutil_test_log_error_counts:n
          { \g_zutil_test_errors_cnt_int - \g_zutil_test_errors_tmp_int }
      }
    % log check counts
    \flag_if_raised:NT \l_zutil_test_checks_cnt_flag
      {
        \int_gadd:Nn \g_zutil_test_check_cnt_int
          { \flag_height:N \l_zutil_test_checks_cnt_flag }
        \zutil_test_log_check_counts:n
          { \flag_height:N \l_zutil_test_checks_cnt_flag }
        \flag_clear:N \l_zutil_test_checks_cnt_flag
      }
  }

\cs_new_protected:Npn \zutil_test_add_expandable_errors_cnt:
  {
    \int_gadd:Nn \g_zutil_test_errors_cnt_int
      { \flag_height:N \l_zutil_test_exp_errors_cnt_flag }
  }

\cs_new_protected:Npn \zutil_test_log_error_counts:n #1
  {
    \zutil_test_type:e { \int_eval:n {#1} ~error(s) }
  }
\cs_new_protected:Npn \zutil_test_log_check_counts:n #1
  {
    \zutil_test_type:e { \int_eval:n {#1} ~check(s) }
  }

% suspend and resume error counting
% similar to \debug_suspend: and \debug_resume:
\cs_new_protected:Npn \zutil_test_suspend_error_count:
  {
    \socket_assign_plug:nn { zutil/test/count-errors } { noop }
    \socket_assign_plug:nn { zutil/test/count-exp-errors } { noop }
  }
\cs_new_protected:Npn \zutil_test_resume_error_count:
  {
    \socket_assign_plug:nn { zutil/test/count-errors } { normal }
    \socket_assign_plug:nn { zutil/test/count-exp-errors } { normal }
  }

% will be let to \PASSED at the beginning of each test
\cs_new:Npn \__zutil_test_PASSED:
  {
    \int_compare:nNnTF
      { \currentgrouplevel } > { \g_zutil_test_group_level_int }
      { \msg_expandable_error:nn { zutil/test } { PASSED } }
      { \flag_raise:N \l_zutil_test_checks_cnt_flag }
  }
\cs_undefine:N \FAILED
